function agregarEquipo() { const serie = document.getElementById("serie").value.trim(); const articulo = document.getElementById("articulo").value.trim(); const estado = document.getElementById("estado").value; if (!serie || !articulo) { alert("Debes ingresar tanto la Serie como el Nombre del Artículo."); return; } equipos.push({ serie, articulo, estado }); const tabla = document.getElementById("tablaEquipos"); const tbody = tabla.querySelector("tbody"); const row = tbody.insertRow(); row.insertCell().textContent = serie; row.insertCell().textContent = articulo; row.insertCell().textContent = estado; const accionCell = row.insertCell(); const btnEliminar = document.createElement("button"); btnEliminar.textContent = "Eliminar"; btnEliminar.style.backgroundColor = "#d9534f"; btnEliminar.style.color = "white"; btnEliminar.style.border = "none"; btnEliminar.style.padding = "5px 10px"; btnEliminar.style.borderRadius = "4px"; btnEliminar.style.cursor = "pointer"; // Acción al eliminar btnEliminar.onclick = function () { const index = row.rowIndex - 1; // ajusta porque la cabecera ocupa la fila 0 equipos.splice(index, 1); // quita del array row.remove(); // elimina la fila if (equipos.length === 0) { tabla.style.display = "none"; } }; accionCell.appendChild(btnEliminar); tabla.style.display = "table"; document.getElementById("serie").value = ""; document.getElementById("articulo").value = ""; document.getElementById("estado").value = "Buen estado"; }
